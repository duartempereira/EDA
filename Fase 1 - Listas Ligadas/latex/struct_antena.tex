\doxysubsection{Detailed Description}
Representa uma antena com uma frequência e coordenadas no mapa. 

Esta estrutura armazena as informações de uma antena, incluindo sua frequência, as coordenadas {\ttfamily x} e {\ttfamily y} e um ponteiro para o próximo elemento da lista ligada.


\begin{DoxyParams}{Parameters}
{\em frequencia} & A frequência da antena, representada por um caractere. \\
\hline
{\em x} & A coordenada X da antena no mapa. \\
\hline
{\em y} & A coordenada Y da antena no mapa. \\
\hline
{\em prox} & Ponteiro para a próxima antena na lista ligada. \\
\hline
\end{DoxyParams}
